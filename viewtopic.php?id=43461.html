    printf("Hello World!\n");
}

Post #2
aport
11 Apr 2013, 04:10

You need to define the prepare stage in your package Makefile:

define Build/Prepare
    mkdir -p $(PKG_BUILD_DIR)
    $(CP) ./src/* $(PKG_BUILD_DIR)/
endef

INSTALL(TARGETS knode RUNTIME DESTINATION sbin)
