One problem: Just after intalling and rebooting, All leds were keeping on lighting. This prevented ethernet port from linking. After "echo 0 > /proc/diag/led/power", ethernet ports linked and leds became normal. So I had to change /etc/diag.sh

root@OpenWrt:/# cat /etc/diag.sh
#!/bin/sh
# Copyright (C) 2006 OpenWrt.org

set_led() {
        local led="$1"
        local state="$2"
        [ -f "/proc/diag/led/$1" ] && echo "$state" > "/proc/diag/led/$1"
}

set_state() {
        case "$1" in
                preinit)
                        set_led dmz 1
                        set_led diag 1
                        set_led power 0
                ;;
                failsafe)
                        set_led diag f
                        set_led power f
                        set_led dmz f
                ;;
                done)
                        set_led dmz 0
                        set_led diag 0
#                       set_led power 1
                        set_led power 0
                ;;
        esac
}
